{{ _("Quickstart (local demo)") }}

git clone https://github.com/elh0m3r0/sme-client-portal.git
cd sme-client-portal
cp .env.example .env
docker compose up --build

# v jiném terminálu — připrav demo data
docker compose exec web python -m scripts.seed_dev

# přihlas se na http://localhost:8000
# owner@4mex.cz / demo1234 (staff)
# jan@acme.cz   / demo1234 (kontakt klienta)
